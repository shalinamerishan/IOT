function doPost(e) {
  var sheet = SpreadsheetApp.openById("YOUR_SHEET_ID").getActiveSheet();
  var data = JSON.parse(e.postData.contents);

  sheet.appendRow([
    new Date(),
    data.temp,
    data.hum,
    data.soil,
    data.light,
    data.fan,
    data.pump
  ]);

  return ContentService.createTextOutput("Success");
}
